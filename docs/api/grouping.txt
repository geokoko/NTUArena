/auth
  POST /register
  POST /login

/tournaments
  POST /                		-> Admin creates tournament
  POST /:id/start       		-> Admin starts tournament
  POST /:id/end         		-> Admin ends tournament
  GET  /:id             		-> Get tournament public info
  GET  /:id/standings   		-> Public standings view
  POST /:id/join        		-> Player joins tournament
  GET  /:id/players     		-> Admin view of players

/players
  GET /:playerId/stats  		-> Player views their stats

/games
  POST /:gameId/submitResult 	-> Admin records game result
  GET /activegames           	-> Admin view of ongoing games
